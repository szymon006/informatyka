#include <iostream>
 
using namespace std;
 
void czy_doskonala(unsigned a)
{
    if(a<1)return;
    int suma=0;
    int iloczyn = 1;
    for(int i=1; i<=a/2; i++)
        if(a%i==0)    {
            suma+=i;
            iloczyn *= i;
        }
 
    cout << endl << "Liczba " << a << ((suma==a) ? "" : " nie") << " jest liczba doskonala " << endl;
}
 
int main()
{
 
    int x;
    do
    {
        cout<<endl;
        cout<<"podaj liczbe";
        cin >> x;
        czy_doskonala(x);
    }while(x!=0);
 
 
 
    system("pause");
    return 0;
}
