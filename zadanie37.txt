#include <cstdlib>
#include <iostream>
#include <conio.h>
using namespace std;
int main(int argc, char *argv[])
{
 int a[7][7];
 for(int i=0;i<6;i++)
 {
 for(int j=0;j<6;j++)
 {
 if(i>=j) {a[i][j]=1;}
 else
 a[i][j]=0;
 }
 }
 for(int i1=0;i1<6;i1++)
 {
 for(int j1=0;j1<6;j1++)
 {
 cout<<a[i1][j1]<<" ";
 }
 cout<<"\n";
 }

 system("PAUSE");
 return EXIT_SUCCESS;
}
